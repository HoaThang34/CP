Tổ hợp chập k của n C(k, n)
long long C(long long k, long long n){
    if (k < 0 || k > n){
        return 0;
    }
    if (k == 0 || k == n){
        return 1;
    }
    if (n / 2 < k){
        k = n - k;
    }
    long long ans = 1;
    for (long long i = 1; i <= k; i++){
        ans = ans *(n - i + 1)/i;
    }
    return ans;
}

vector<vector<long long>> d;
void bfs(long long sx,long long sy){
    queue<pair<long long,long long>> q;
    q.push({sx,sy});
    d[sx][sy]=0;
    while(!q.empty()){
        pair<long long,long long> u=q.front();q.pop();
        for(int k=0;k<4;k++){
            long long x=u.first+dx[k];
            long long y=u.second+dy[k];

                d[x][y]=d[u.first][u.second]+1;
                q.push({x,y});
            }
        }
    }
}

Ước chung lớn nhất (GCD):

long long gcd(long long a, long long b){
    long long tmp;
    while (b > 0){
        tmp = a % b;
        a = b;
        b = tmp;
    }
    return a;
}

 Segment tree (truy vấn tìm min trên đoạn [l, r]);
struct SegTree {
    int n;
    vector<int> st;

    SegTree(int n) : n(n) {
        st.assign(4*n, INT_MAX);
    }

    void build(vector<int> &a, int id, int l, int r) {
        if (l == r) {
            st[id] = a[l];
            return;
        }
        int mid = (l + r) / 2;
        build(a, id*2, l, mid);
        build(a, id*2+1, mid+1, r);
        st[id] = min(st[id*2], st[id*2+1]);
    }

    int query(int id, int l, int r, int u, int v) {
        if (v < l || r < u) return INT_MAX;
        if (u <= l && r <= v) return st[id];
        int mid = (l + r) / 2;
        int leftVal = query(id*2, l, mid, u, v);
        int rightVal = query(id*2+1, mid+1, r, u, v);
        return min(leftVal, rightVal);
    }
};


Tổng hợp segment tree

#include <bits/stdc++.h>
using namespace std;

struct SegTree {
    struct Node {
        long long sum = 0; 
        long long mx = LLONG_MIN;
        long long lazy = 0;     // lazy add
    };

    int n;
    vector<Node> st;

    SegTree(int n) : n(n) {
        st.resize(4*n);
    }

    void build(vector<long long> &a, int id, int l, int r) {
        if (l == r) {
            st[id].sum = st[id].mx = a[l];
            return;
        }
        int mid = (l + r) / 2;
        build(a, id*2, l, mid);
        build(a, id*2+1, mid+1, r);
        pull(id);
    }

    void push(int id, int l, int r) {
        long long &lz = st[id].lazy;
        if (lz == 0) return;

        st[id].sum += lz * (r - l + 1);
        st[id].mx += lz;

        if (l != r) {
            st[id*2].lazy += lz;
            st[id*2+1].lazy += lz;
        }
        lz = 0;
    }

    void pull(int id) {
        st[id].sum = st[id*2].sum + st[id*2+1].sum;
        st[id].mx = max(st[id*2].mx, st[id*2+1].mx);
    }

    void update(int id, int l, int r, int u, int v, long long val) {
        push(id, l, r);
        if (v < l || r < u) return;
        if (u <= l && r <= v) {
            st[id].lazy += val;
            push(id, l, r);
            return;
        }
        int mid = (l + r) / 2;
        update(id*2, l, mid, u, v, val);
        update(id*2+1, mid+1, r, u, v, val);
        pull(id);
    }

    long long query_sum(int id, int l, int r, int u, int v) {
        push(id, l, r);
        if (v < l || r < u) return 0;
        if (u <= l && r <= v) return st[id].sum;
        int mid = (l + r) / 2;
        return query_sum(id*2, l, mid, u, v)
             + query_sum(id*2+1, mid+1, r, u, v);
    }

    long long query_max(int id, int l, int r, int u, int v) {
        push(id, l, r);
        if (v < l || r < u) return LLONG_MIN;
        if (u <= l && r <= v) return st[id].mx;
        int mid = (l + r) / 2;
        return max(
            query_max(id*2, l, mid, u, v),
            query_max(id*2+1, mid+1, r, u, v)
        );
    }
};
